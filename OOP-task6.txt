CLASS WATER_BILL EXTENDS BILL
    ATTRIBUTES
        LITERS_CONSUMED: NUMBER
        FLAT_RATE: NUMBER
        STANDING_CHARGE: NUMBER


        CONSTRUCTOR(billingPeriod: TEXT, dueDate: DATE, meterReading: METER_READING, litersConsumed: NUMBER,    \
        flatRate:  
        NUMBER, standingCharge: NUMBER)
        CALL BILL.CONSTRUCTOR(billingPeriod, dueDate, meterReading)
        SET LITERS_CONSUMED TO litersConsumed
        SET FLAT_RATE TO flatRate
        SET STANDING_CHARGE TO standingCharge
    END CONSTRUCTOR
    
        CALCULATE_AMOUNT()
        RETURN (LITERS_CONSUMED * FLAT_RATE) + STANDING_CHARGE
     END METHOD
  END METHODS
END CLASS



CLASS GAS_BILL EXTENDS BILL
    ATTRIBUTES
        UNITS_CONSUMED: NUMBER
        RATE: NUMBER
        SEASON: TEXT
    CONSTRUCTOR(billingPeriod: TEXT, dueDate: DATE, meterReading: METER_READING, unitsConsumed: NUMBER, rate: NUMBER, season: TEXT)
        CALL BILL.CONSTRUCTOR(billingPeriod, dueDate, meterReading)
        SET UNITS_CONSUMED TO unitsConsumed
        SET RATE TO rate
        SET SEASON TO season
    END CONSTRUCTOR
    METHODS
        CALCULATE_AMOUNT()
            SET TOTAL TO UNITS_CONSUMED * RATE
            IF SEASON EQUALS "WINTER" THEN
            SET TOTAL TO TOTAL + (TOTAL * 0.10)
         END IF
            RETURN TOTAL
       END METHOD
    END METHODS
END CLASS

        CALL RIDE.CONSTRUCTOR(pickupLocation, dropOffLocation, route)

        SET BASE_RATE TO baseRate
        SET COMFORT_SURCHARGE TO comfortSurcharge

CLASS SHARED_RIDE EXTENDS STANDARD_RIDE
    ATTRIBUTES
        PASSENGERS: NUMBER
    CONSTRUCTOR(pickupLocation: TEXT, dropOffLocation: TEXT, route: ROUTE, distance: NUMBER, baseRate: NUMBER, passengers: NUMBER)
        CALL STANDARD_RIDE.CONSTRUCTOR(pickupLocation, dropOffLocation, route, distance, baseRate)
        SET PASSENGERS TO passengers
    END CONSTRUCTOR
    METHODS
        CALCULATE_FARE()
            SET TOTAL TO CALL STANDARD_RIDE.CALCULATE_FARE()
            RETURN TOTAL / PASSENGERS
     END METHOD
  END METHODS
END CLASS
