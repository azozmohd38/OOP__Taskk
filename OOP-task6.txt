
Task 303: Musical Instruments Store  



CLASS MUSIC_STORE
    ATTRIBUTES
        INSTRUMENTS: LIST<INSTRUMENT>

    CONSTRUCTOR()
        SET INSTRUMENTS TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_INSTRUMENT(instrument: INSTRUMENT)
        ADD instrument TO INSTRUMENTS
      END METHOD

        DISPLAY_INVENTORY()
            FOR EACH instrument IN INSTRUMENTS
                DISPLAY instrument.BRAND
                DISPLAY instrument.PRICE
                DISPLAY instrument.PLAY_DESCRIPTION()
        END FOR
     END METHOD
  END METHODS
END CLASS


CLASS INSTRUMENT
    ATTRIBUTES
        BRAND: TEXT
        PRICE: NUMBER

    CONSTRUCTOR(brand: TEXT, price: NUMBER)
        SET BRAND TO brand
        SET PRICE TO price
    END CONSTRUCTOR

    METHODS
        PLAY_DESCRIPTION()
            RETURN "Instrument is played."
     END METHOD
  END METHODS
END CLASS


CLASS GUITAR INHERITS INSTRUMENT
    ATTRIBUTES
        STRINGS: NUMBER

    CONSTRUCTOR(brand: TEXT, price: NUMBER, strings: NUMBER)
        SUPER(brand, price)
        SET STRINGS TO strings
    END CONSTRUCTOR

    METHODS
        PLAY_DESCRIPTION()
        RETURN "Strummed or plucked."
     END METHOD
   END METHODS
END CLASS


CLASS DRUM_KIT INHERITS INSTRUMENT
    ATTRIBUTES
        DRUM_PIECES: NUMBER

    CONSTRUCTOR(brand: TEXT, price: NUMBER, drumPieces: NUMBER)
        SUPER(brand, price)
        SET DRUM_PIECES TO drumPieces
    END CONSTRUCTOR

    METHODS
        PLAY_DESCRIPTION()
        RETURN "Struck."
     END METHOD
   END METHODS
END CLASS


CLASS KEYBOARD INHERITS INSTRUMENT
    ATTRIBUTES
        KEYS: NUMBER

    CONSTRUCTOR(brand: TEXT, price: NUMBER, keys: NUMBER)
        SUPER(brand, price)
        SET KEYS TO keys
    END CONSTRUCTOR

    METHODS
        PLAY_DESCRIPTION()
        RETURN "Pressed."
     END METHOD
   END METHODS
END CLASS


START
    DECLARE store AS MUSIC_STORE
    SET store TO NEW MUSIC_STORE()

    DECLARE guitar AS GUITAR
    SET guitar TO NEW GUITAR("Yamaha", 500, 6)

    DECLARE drumKit AS DRUM_KIT
    SET drumKit TO NEW DRUM_KIT("Pearl", 800, 5)

    DECLARE keyboard AS KEYBOARD
    SET keyboard TO NEW KEYBOARD("Casio", 300, 61)

    CALL store.ADD_INSTRUMENT(guitar)
    CALL store.ADD_INSTRUMENT(drumKit)
    CALL store.ADD_INSTRUMENT(keyboard)
    CALL store.DISPLAY_INVENTORY()
END START





Task 304: Fitness App Workouts  


CLASS FITNESS_LOG
    ATTRIBUTES
        WORKOUTS: LIST<WORKOUT>

    CONSTRUCTOR()
        SET WORKOUTS TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_WORKOUT(workout: WORKOUT)
        ADD workout TO WORKOUTS
      END METHOD

        DISPLAY_TOTAL_CALORIES()
            DECLARE TOTAL AS NUMBER
            SET TOTAL TO 0

            FOR EACH workout IN WORKOUTS
            SET TOTAL TO TOTAL + workout.CALCULATE_CALORIES()
          END FOR
            DISPLAY "Total Calories Burned: ", TOTAL
      END METHOD
   END METHODS
END CLASS


CLASS WORKOUT
    ATTRIBUTES
        DATE: DATE
        DURATION: NUMBER

    CONSTRUCTOR(date: DATE, duration: NUMBER)
        SET DATE TO date
        SET DURATION TO duration
    END CONSTRUCTOR

    METHODS
        CALCULATE_CALORIES()
        RETURN 0
      END METHOD
   END METHODS
END CLASS


CLASS RUNNING_WORKOUT INHERITS WORKOUT
    ATTRIBUTES
        DISTANCE: NUMBER
        PACE: NUMBER

    CONSTRUCTOR(date: DATE, duration: NUMBER, distance: NUMBER, pace: NUMBER)
        SUPER(date, duration)
        SET DISTANCE TO distance
        SET PACE TO pace
    END CONSTRUCTOR

    METHODS
        CALCULATE_CALORIES()
            Calories = Distance × Pace
            RETURN DISTANCE * PACE
      END METHOD
   END METHODS
END CLASS


CLASS CYCLING_WORKOUT INHERITS WORKOUT
    ATTRIBUTES
        DISTANCE: NUMBER
        RESISTANCE_LEVEL: NUMBER

    CONSTRUCTOR(date: DATE, duration: NUMBER, distance: NUMBER, resistanceLevel: NUMBER)
        SUPER(date, duration)
        SET DISTANCE TO distance
        SET RESISTANCE_LEVEL TO resistanceLevel
    END CONSTRUCTOR

    METHODS
        CALCULATE_CALORIES()
            Calories = Distance × Resistance Level
            RETURN DISTANCE * RESISTANCE_LEVEL
     END METHOD
  END METHODS
END CLASS


CLASS WEIGHT_TRAINING_WORKOUT INHERITS WORKOUT
    ATTRIBUTES
        SETS: NUMBER
        REPS: NUMBER

    CONSTRUCTOR(date: DATE, duration: NUMBER, sets: NUMBER, reps: NUMBER)
        SUPER(date, duration)
        SET SETS TO sets
        SET REPS TO reps
    END CONSTRUCTOR

    METHODS
        CALCULATE_CALORIES()
            Calories = Sets × Reps
            RETURN SETS * REPS
    END METHOD
  END METHODS
END CLASS


START
    DECLARE log AS FITNESS_LOG
    SET log TO NEW FITNESS_LOG()

    DECLARE run AS RUNNING_WORKOUT
    SET run TO NEW RUNNING_WORKOUT("2026-07-21", 30, 5, 80)

    DECLARE cycle AS CYCLING_WORKOUT
    SET cycle TO NEW CYCLING_WORKOUT("2026-07-22", 45, 15, 6)

    DECLARE weights AS WEIGHT_TRAINING_WORKOUT
    SET weights TO NEW WEIGHT_TRAINING_WORKOUT("2026-07-23", 60, 4, 12)

    CALL log.ADD_WORKOUT(run)
    CALL log.ADD_WORKOUT(cycle)
    CALL log.ADD_WORKOUT(weights)
    CALL log.DISPLAY_TOTAL_CALORIES()
END START

    END METHOD

    END CONSTRUCTOR

    METHODS

        END METHOD

CLASS CHARACTER
    ATTRIBUTES
        NAME: TEXT
        HEALTH: NUMBER
        INVENTORY: INVENTORY

    CONSTRUCTOR(name: TEXT, health: NUMBER)
        SET NAME TO name
        SET HEALTH TO health
        SET INVENTORY TO NEW INVENTORY()
    END CONSTRUCTOR

    METHODS
        ATTACK()
        RETURN "CHARACTER ATTACKS"
    END METHOD
  END METHODS

CLASS WARRIOR INHERITS CHARACTER
    ATTRIBUTES
        HAS_HEAVY_ARMOR: BOOLEAN

    CONSTRUCTOR(name: TEXT, health: NUMBER, hasHeavyArmor: BOOLEAN)
        SUPER(name, health)
        SET HAS_HEAVY_ARMOR TO hasHeavyArmor
    END CONSTRUCTOR

    METHODS
        ATTACK()
        RETURN "ATTACK USING MELEE STRENGTH"
    END METHOD
  END METHODS
END CLASS


CLASS MAGE INHERITS CHARACTER
    ATTRIBUTES
        MANA: NUMBER

    CONSTRUCTOR(name: TEXT, health: NUMBER, mana: NUMBER)
        SUPER(name, health)
        SET MANA TO mana
    END CONSTRUCTOR

    METHODS
        ATTACK()
            IF MANA GREATER THAN 0 THEN
            SET MANA TO MANA - 1
            RETURN "ATTACK USING MAGIC SPELL"
ELSE
            RETURN "NO MANA"
        END IF
     END METHOD
  END METHODS
END CLASS


CLASS ARCHER INHERITS CHARACTER
    ATTRIBUTES
        ARROWS: NUMBER

    CONSTRUCTOR(name: TEXT, health: NUMBER, arrows: NUMBER)
        SUPER(name, health)
        SET ARROWS TO arrows
    END CONSTRUCTOR

    METHODS
        ATTACK()
            IF ARROWS GREATER THAN 0 THEN
            SET ARROWS TO ARROWS - 1
            RETURN "ATTACK USING ARROW"
ELSE
            RETURN "NO ARROWS"
       END IF
    END METHOD
END CLASS


START
    DECLARE sword AS ITEM
    SET sword TO NEW ITEM("Sword")

    DECLARE staff AS ITEM
    SET staff TO NEW ITEM("Magic Staff")

    DECLARE bow AS ITEM
    SET bow TO NEW ITEM("Bow")

    DECLARE warrior AS WARRIOR
    SET warrior TO NEW WARRIOR("Arthur", 100, TRUE)

    DECLARE mage AS MAGE
    SET mage TO NEW MAGE("Merlin", 80, 3)

    DECLARE archer AS ARCHER
    SET archer TO NEW ARCHER("Robin", 90, 5)

    CALL warrior.INVENTORY.ADD_ITEM(sword)
    CALL mage.INVENTORY.ADD_ITEM(staff)
    CALL archer.INVENTORY.ADD_ITEM(bow)

    DISPLAY warrior.ATTACK()
    DISPLAY mage.ATTACK()
    DISPLAY archer.ATTACK()

    DISPLAY "Warrior Inventory: "
    CALL warrior.INVENTORY.DISPLAY_ITEMS()

    DISPLAY "Mage Inventory: "
    CALL mage.INVENTORY.DISPLAY_ITEMS()

    DISPLAY "Archer Inventory: "
    CALL archer.INVENTORY.DISPLAY_ITEMS()
END START




Task 306: Utility Billing System  





CLASS CUSTOMER_ACCOUNT
    ATTRIBUTES
        BILLS: LIST<BILL>

    CONSTRUCTOR()
        SET BILLS TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_BILL(bill: BILL)
        ADD bill TO BILLS
     END METHOD

        DISPLAY_TOTAL_AMOUNT()
            DECLARE TOTAL AS NUMBER
            SET TOTAL TO 0

            FOR EACH bill IN BILLS
            SET TOTAL TO TOTAL + bill.CALCULATE_AMOUNT()
          END FOR
            DISPLAY "Total Amount Due: ", TOTAL
      END METHOD
   END METHODS
END CLASS


CLASS METER_READING
    ATTRIBUTES
        START_READING: NUMBER
        END_READING: NUMBER

    CONSTRUCTOR(startReading: NUMBER, endReading: NUMBER)
        SET START_READING TO startReading
        SET END_READING TO endReading
    END CONSTRUCTOR

    METHODS
        GET_CONSUMPTION()
        RETURN END_READING - START_READING
     END METHOD
   END METHODS
END CLASS


CLASS BILL
    ATTRIBUTES
        BILLING_PERIOD: TEXT
        DUE_DATE: DATE
        METER_READING: METER_READING

    CONSTRUCTOR(billingPeriod: TEXT, dueDate: DATE, meterReading: METER_READING)
        SET BILLING_PERIOD TO billingPeriod
        SET DUE_DATE TO dueDate
        SET METER_READING TO meterReading
    END CONSTRUCTOR

    METHODS
        CALCULATE_AMOUNT()
        RETURN 0
     END METHOD
   END METHODS
END CLASS


CLASS ELECTRICITY_BILL INHERITS BILL
    ATTRIBUTES
        PEAK_RATE: NUMBER
        OFF_PEAK_RATE: NUMBER
        IS_PEAK_TIME: BOOLEAN

    CONSTRUCTOR(billingPeriod: TEXT, dueDate: DATE, meterReading: METER_READING,
                peakRate: NUMBER, offPeakRate: NUMBER, isPeakTime: BOOLEAN)

        SUPER(billingPeriod, dueDate, meterReading)
        SET PEAK_RATE TO peakRate
        SET OFF_PEAK_RATE TO offPeakRate
        SET IS_PEAK_TIME TO isPeakTime
    END CONSTRUCTOR

    METHODS
        CALCULATE_AMOUNT()

            IF IS_PEAK_TIME THEN
            RETURN METER_READING.GET_CONSUMPTION() * PEAK_RATE
ELSE
            RETURN METER_READING.GET_CONSUMPTION() * OFF_PEAK_RATE
       END IF
     END METHOD
   END METHODS
END CLASS


CLASS WATER_BILL INHERITS BILL
    ATTRIBUTES
        FLAT_RATE: NUMBER
        STANDING_CHARGE: NUMBER

    CONSTRUCTOR(billingPeriod: TEXT, dueDate: DATE, meterReading: METER_READING,
                flatRate: NUMBER, standingCharge: NUMBER)

        SUPER(billingPeriod, dueDate, meterReading)
        SET FLAT_RATE TO flatRate
        SET STANDING_CHARGE TO standingCharge
    END CONSTRUCTOR

    METHODS
        CALCULATE_AMOUNT()

            RETURN (METER_READING.GET_CONSUMPTION() * FLAT_RATE)+ STANDING_CHARGE
      END METHOD
  END METHODS
END CLASS


CLASS GAS_BILL INHERITS BILL
    ATTRIBUTES
        RATE: NUMBER
        SEASON: TEXT

    CONSTRUCTOR(billingPeriod: TEXT, dueDate: DATE, meterReading: METER_READING,rate: NUMBER, season: TEXT)

        SUPER(billingPeriod, dueDate, meterReading)
        SET RATE TO rate
        SET SEASON TO season
    END CONSTRUCTOR

    METHODS
        CALCULATE_AMOUNT()
            DECLARE TOTAL AS NUMBER
            SET TOTAL TO METER_READING.GET_CONSUMPTION() * RATE

            IF SEASON EQUALS "WINTER" THEN
            SET TOTAL TO TOTAL + (TOTAL * 0.10)
         END IF
            RETURN TOTAL
      END METHOD
  END METHODS
END CLASS


START

    DECLARE account AS CUSTOMER_ACCOUNT
    SET account TO NEW CUSTOMER_ACCOUNT()


    DECLARE electricityReading AS METER_READING
    SET electricityReading TO NEW METER_READING(1000, 1200)


    DECLARE waterReading AS METER_READING
    SET waterReading TO NEW METER_READING(500, 650)


    DECLARE gasReading AS METER_READING
    SET gasReading TO NEW METER_READING(300, 420)


    DECLARE electricity AS ELECTRICITY_BILL
    SET electricity TO NEW ELECTRICITY_BILL("July","2026-08-10",electricityReading,0.20,0.15,TRUE)


    DECLARE water AS WATER_BILL
    SET water TO NEW WATER_BILL("July","2026-08-10",waterReading,0.05,10)


    DECLARE gas AS GAS_BILL
    SET gas TO NEW GAS_BILL("July","2026-08-10",gasReading,0.18,"WINTER")


    CALL account.ADD_BILL(electricity)
    CALL account.ADD_BILL(water)
    CALL account.ADD_BILL(gas)
    CALL account.DISPLAY_TOTAL_AMOUNT()

END START


Task 307: Ride-Sharing Vehicle Tiers  




CLASS ROUTE
    ATTRIBUTES
        WAYPOINTS: LIST OF WAYPOINT

    CONSTRUCTOR()
        SET WAYPOINTS TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_WAYPOINT(waypoint: WAYPOINT)
            ADD waypoint TO WAYPOINTS
        END METHOD
    END METHODS
END CLASS



CLASS WAYPOINT
    ATTRIBUTES
        LOCATION: TEXT

    CONSTRUCTOR(location: TEXT)
    SET LOCATION TO location
   END CONSTRUCTOR

    METHODS
        GET_LOCATION()
        RETURN LOCATION
    END METHOD
  END METHODS
END CLASS



CLASS RIDE
    ATTRIBUTES
        PICKUP_LOCATION: TEXT
        DROP_OFF_LOCATION: TEXT
        ROUTE: ROUTE

    CONSTRUCTOR(pickupLocation: TEXT, dropOffLocation: TEXT, route: ROUTE)
        SET PICKUP_LOCATION TO pickupLocation
        SET DROP_OFF_LOCATION TO dropOffLocation
        SET ROUTE TO route
    END CONSTRUCTOR

    METHODS
        CALCULATE_FARE()
        RETURN 0
    END METHOD
  END METHODS
END CLASS



CLASS STANDARD_RIDE INHERITS RIDE
    ATTRIBUTES
        DISTANCE: NUMBER
        BASE_RATE: NUMBER

    CONSTRUCTOR(pickupLocation: TEXT, dropOffLocation: TEXT,
                route: ROUTE, distance: NUMBER, baseRate: NUMBER)

        SUPER(pickupLocation, dropOffLocation, route)
        SET DISTANCE TO distance
        SET BASE_RATE TO baseRate

    END CONSTRUCTOR

    METHODS
        CALCULATE_FARE()
        RETURN DISTANCE * BASE_RATE
     END METHOD
   END METHODS
END CLASS



CLASS PREMIUM_RIDE INHERITS RIDE
    ATTRIBUTES
        DISTANCE: NUMBER
        BASE_RATE: NUMBER
        COMFORT_SURCHARGE: NUMBER

    CONSTRUCTOR(pickupLocation: TEXT, dropOffLocation: TEXT,
                route: ROUTE, distance: NUMBER,
                baseRate: NUMBER, comfortSurcharge: NUMBER)

        SUPER(pickupLocation, dropOffLocation, route)
        SET DISTANCE TO distance
        SET BASE_RATE TO baseRate
        SET COMFORT_SURCHARGE TO comfortSurcharge

    END CONSTRUCTOR

    METHODS
        CALCULATE_FARE()
        RETURN (DISTANCE * BASE_RATE) + COMFORT_SURCHARGE
     END METHOD
   END METHODS
END CLASS



CLASS SHARED_RIDE INHERITS STANDARD_RIDE
    ATTRIBUTES
        PASSENGERS: NUMBER

    CONSTRUCTOR(pickupLocation: TEXT, dropOffLocation: TEXT,route: ROUTE, distance: NUMBER,baseRate: NUMBER, passengers: NUMBER)
        SUPER(pickupLocation, dropOffLocation, route, distance, baseRate)
        SET PASSENGERS TO passengers

    END CONSTRUCTOR

    METHODS
        CALCULATE_FARE()

            DECLARE TOTAL AS NUMBER
            SET TOTAL TO SUPER.CALCULATE_FARE()
            RETURN TOTAL / PASSENGERS
     END METHOD
   END METHODS
END CLASS



START

    DECLARE route AS ROUTE
    SET route TO NEW ROUTE()


    DECLARE waypoint1 AS WAYPOINT
    SET waypoint1 TO NEW WAYPOINT("Airport")


    DECLARE waypoint2 AS WAYPOINT
    SET waypoint2 TO NEW WAYPOINT("Downtown")


    DECLARE waypoint3 AS WAYPOINT
    SET waypoint3 TO NEW WAYPOINT("Hotel")


    CALL route.ADD_WAYPOINT(waypoint1)
    CALL route.ADD_WAYPOINT(waypoint2)
    CALL route.ADD_WAYPOINT(waypoint3)


    DECLARE standardRide AS STANDARD_RIDE
    SET standardRide TO NEW STANDARD_RIDE("Airport","Hotel",route,20,2)


    DECLARE premiumRide AS PREMIUM_RIDE
    SET premiumRide TO NEW PREMIUM_RIDE("Airport","Hotel",route,20,2,10)
    DECLARE sharedRide AS SHARED_RIDE
    SET sharedRide TO NEW SHARED_RIDE("Airport","Hotel",route,20,2,4)
    DISPLAY "Standard Ride Fare: ", standardRide.CALCULATE_FARE()
    DISPLAY "Premium Ride Fare: ", premiumRide.CALCULATE_FARE()
    DISPLAY "Shared Ride Fare: ", sharedRide.CALCULATE_FARE()


END START
