CLASS MEDICAL_CLAIM EXTENDS CLAIM
    ATTRIBUTES
        TREATMENT_COST: NUMBER
        COVERAGE_LIMIT: NUMBER
    CONSTRUCTOR(claimNumber: TEXT, dateFiled: DATE, status: TEXT, treatmentCost: NUMBER, coverageLimit: NUMBER)
        CALL CLAIM.CONSTRUCTOR(claimNumber, dateFiled, status)
        SET TREATMENT_COST TO treatmentCost
        SET COVERAGE_LIMIT TO coverageLimit
    END CONSTRUCTOR
    METHODS
        CALCULATE_PAYOUT()
            IF TREATMENT_COST LESS THAN COVERAGE_LIMIT THEN
            REPEAT COVERAGE_LIMIT
ELSE
            REPEAT TREATMENT_COST
         END IF
    END METHOD
  END METHODS
END CLASS



CLASS PROPERTY_CLAIM EXTENDS CLAIM
    ATTRIBUTES
        DAMAGE_VALUE: NUMBER
        MAINTENANCE_REDUCTION: NUMBER
    CONSTRUCTOR(claimNumber: TEXT, dateFiled: DATE, status: TEXT, damageValue: NUMBER, maintenanceReduction:  
       NUMBER)
        CALL CLAIM.CONSTRUCTOR(claimNumber, dateFiled, status)
        SET DAMAGE_VALUE TO damageValue
        SET MAINTENANCE_REDUCTION TO maintenanceReduction
    END CONSTRUCTOR
    METHODS
        CALCULATE_PAYOUT()
        REPEAT DAMAGE_VALUE - (DAMAGE_VALUE * MAINTENANCE_REDUCTION / 100)
      END METHOD
   END METHODS
END CLASS

CLASS VIP_ACCESS_RIDE EXTENDS ROLLER_COASTER
    ATTRIBUTES
        VIP_GUESTS: NUMBER
        STANDARD_GUESTS: NUMBER

      NUMBER, rideCycleTime: NUMBER, vipGuests: NUMBER, standardGuests: NUMBER)

             rideCycleTime)
        SET VIP_GUESTS TO vipGuests

    END CONSTRUCTOR

        RETURN (STANDARD_GUESTS / CAR_CAPACITY) * RIDE_CYCLE_TIME
      END METHOD

        REPEAT (VIP_GUESTS / CAR_CAPACITY) * RIDE_CYCLE_TIME
      END METHOD
    END METHODS
END CLASS
