        SET REPAIR_ESTIMATE TO repairEstimate

    END CONSTRUCTOR

    METHODS

CLASS MEDICAL_CLAIM INHERITS CLAIM
    ATTRIBUTES
        TREATMENT_COST: NUMBER
        COVERAGE_LIMIT: NUMBER

    CONSTRUCTOR(claimNumber: TEXT, dateFiled: DATE, status: TEXT,
                treatmentCost: NUMBER, coverageLimit: NUMBER)
        CALL SUPER(claimNumber, dateFiled, status)
        SET TREATMENT_COST TO treatmentCost
        SET COVERAGE_LIMIT TO coverageLimit
    END CONSTRUCTOR

    METHODS
        CALCULATE_PAYOUT()
            IF TREATMENT_COST LESS THAN OR EQUAL TO COVERAGE_LIMIT THEN
            RETURN TREATMENT_COST
ELSE
            RETURN COVERAGE_LIMIT
       END IF
    END METHODS
END CLASS


CLASS PROPERTY_CLAIM INHERITS CLAIM
    ATTRIBUTES

    CONSTRUCTOR(claimNumber: TEXT, dateFiled: DATE, status: TEXT,
                damageValue: NUMBER, maintenanceReduction: NUMBER)

        SET DAMAGE_VALUE TO damageValue
        SET MAINTENANCE_REDUCTION TO maintenanceReduction
    END CONSTRUCTOR

    METHODS
        CALCULATE_PAYOUT()
        RETURN DAMAGE_VALUE - (DAMAGE_VALUE * MAINTENANCE_REDUCTION / 100)
     END METHOD
  END METHODS
END CLASS

   END METHODS

CLASS DATABASE_SERVER INHERITS SERVER
    ATTRIBUTES
        QUERY_THROUGHPUT: NUMBER
        MAX_QUERIES_PER_SECOND: NUMBER

    CONSTRUCTOR(serverId: TEXT, serialNumber: TEXT,uptime: NUMBER, cpu: TEXT,ram: TEXT, storage: TEXT,queryThroughput: NUMBER,maxQueriesPerSecond: NUMBER)

        CALL SUPER(serverId, serialNumber, uptime,cpu, ram, storage)
        SET QUERY_THROUGHPUT TO queryThroughput
        SET MAX_QUERIES_PER_SECOND TO maxQueriesPerSecond

    END CONSTRUCTOR

    METHODS
        CALCULATE_LOAD()
        RETURN (QUERY_THROUGHPUT / MAX_QUERIES_PER_SECOND) * 100
       END METHOD

        ACCEPT_CONNECTION()
            IF QUERY_THROUGHPUT LESS THAN MAX_QUERIES_PER_SECOND THEN
            RETURN "Connection Accepted"
ELSE
           RETURN "Connection Refused"
       END IF
     END METHOD
   END METHODS
END CLASS



CLASS CACHE_SERVER INHERITS DATABASE_SERVER
    ATTRIBUTES
        CACHE_HIT_RATIO: NUMBER

    CONSTRUCTOR(serverId: TEXT, serialNumber: TEXT,uptime: NUMBER, cpu: TEXT, ram: TEXT, storage: NUMBER,queryThroughput: NUMBER,maxQueriesPerSecond: NUMBER, cacheHitRatio: NUMBER)

        CALL SUPER(serverId, serialNumber, uptime,cpu, ram, storage,queryThroughput,maxQueriesPerSecond)
        SET CACHE_HIT_RATIO TO cacheHitRatio

    END CONSTRUCTOR

    METHODS
        CALCULATE_LOAD()
        RETURN ((QUERY_THROUGHPUT / MAX_QUERIES_PER_SECOND) * 100)- CACHE_HIT_RATIO)
     END METHOD
   END METHODS
END CLASS
