CLASS DEBIT_CARD INHERITS CARD
    ATTRIBUTES
        LINKED_ACCOUNT: ACCOUNT
    CONSTRUCTOR(cardNumber: TEXT, expiryDate: TEXT, chip: CHIP, account: ACCOUNT)
        CALL CARD.CONSTRUCTOR(cardNumber, expiryDate, chip)
        SET LINKED_ACCOUNT TO account
    END CONSTRUCTOR
    METHODS
        AUTHORIZE_PURCHASE(amount: NUMBER)
            IF LINKED_ACCOUNT.BALANCE GREATER THAN OR EQUALS amount
            RETURN "AUTHORIZED"
ELSE
            RETURN "DECLINED"
      END IF
    END METHOD
  END METHODS
END CLASS



CLASS CREDIT_CARD INHERITS CARD
    ATTRIBUTES
        CREDIT_LIMIT: NUMBER
        PURCHASES: LIST<NUMBER>
    CONSTRUCTOR(cardNumber: TEXT, expiryDate: TEXT, chip: CHIP, limit: NUMBER)
        CALL CARD.CONSTRUCTOR(cardNumber, expiryDate, chip)
        SET CREDIT_LIMIT TO limit
        SET PURCHASES TO EMPTY LIST

    END CONSTRUCTOR

    METHODS

        AUTHORIZE_PURCHASE(amount: NUMBER)
            IF CALCULATE_OUTSTANDING_BALANCE() + amount LESS THAN OR EQUALS CREDIT_LIMIT
            ADD amount TO PURCHASES
            RETURN "AUTHORIZED"
  ELSE
            RETURN "DECLINED"
          END IF
        END METHOD

        CALCULATE_OUTSTANDING_BALANCE()
            SET TOTAL TO 0
            SET i TO 0
            REPEAT
            SET TOTAL TO TOTAL + PURCHASES[i]
            SET i TO i + 1
            IF i EQUAL THE LENGTH(PURCHASES)
            RETURN TOTAL
    END METHOD
  END METHODS
END CLASS
