CLASS CREDIT_CARD INHERITS CARD

    ATTRIBUTES
        CREDIT_LIMIT: NUMBER
        PURCHASES: LIST<NUMBER>


    CONSTRUCTOR(cardNumber: TEXT, expiryDate: TEXT, chip: CHIP, limit: NUMBER)

        CALL SUPER(cardNumber, expiryDate, chip)

        SET THIS.CREDIT_LIMIT TO limit
        SET THIS.PURCHASES TO EMPTY LIST

    END CONSTRUCTOR



    METHODS


        AUTHORIZE_PURCHASE(amount: NUMBER)

            IF THIS.STATUS EQUALS "ACTIVE" AND amount GREATER THAN 0 AND THIS.CALCULATE_OUTSTANDING_BALANCE() + amount LESS THAN OR EQUAL TO THIS.CREDIT_LIMIT THEN

                ADD amount TO THIS.PURCHASES

                RETURN "AUTHORIZED"

            ELSE

                RETURN "DECLINED"

            END IF

        END METHOD



        CALCULATE_OUTSTANDING_BALANCE()

            SET TOTAL TO 0
            SET i TO 0

            REPEAT

                SET TOTAL TO TOTAL + THIS.PURCHASES[i]
                SET i TO i + 1

            UNTIL i EQUALS SIZE OF THIS.PURCHASES

            RETURN TOTAL

        END METHOD



        MAKE_PAYMENT(amount: NUMBER)

            IF amount GREATER THAN 0 THEN

                SET REMAINING_PAYMENT TO amount

                REPEAT

                    IF SIZE OF THIS.PURCHASES GREATER THAN 0 THEN

                        REMOVE FIRST ITEM FROM THIS.PURCHASES

                    ELSE

                        BREAK

                    END IF

                    SET REMAINING_PAYMENT TO REMAINING_PAYMENT - amount

                UNTIL REMAINING_PAYMENT LESS THAN OR EQUAL TO 0 OR SIZE OF THIS.PURCHASES EQUALS 0

            END IF

        END METHOD


    END METHODS

END CLASS

                    SET OPENED_COUNT TO OPENED_COUNT + 1
                END IF
            END FOR

            RETURN (OPENED_COUNT / SIZE OF ARTICLES) * 100
        END METHOD
    END METHODS
END CLASS

CLASS STORAGE_UNIT
    ATTRIBUTES
        USED_SPACE: NUMBER
        MAX_SPACE: NUMBER

    CONSTRUCTOR(maxSpace: NUMBER)
        SET USED_SPACE TO 0
        SET MAX_SPACE TO maxSpace
    END CONSTRUCTOR

    METHODS
        CHECK_SPACE()
            RETURN USED_SPACE < MAX_SPACE
        END METHOD

        ADD_DATA(size: NUMBER)
            IF USED_SPACE + size <= MAX_SPACE THEN
                SET USED_SPACE TO USED_SPACE + size
            END IF
        END METHOD
    END METHODS
END CLASS

    CONSTRUCTOR(name: TEXT, location: TEXT, storage: STORAGE_UNIT)
        CALL SUPER(name, location)
        SET IS_RECORDING TO FALSE
        SET STORAGE TO storage
    END CONSTRUCTOR

    METHODS
        START_RECORDING()
            IF STORAGE.CHECK_SPACE() THEN
                SET IS_RECORDING TO TRUE
                CALL STORAGE.ADD_DATA(1)
            END IF
        END METHOD

        REPORT_POWER_USAGE()
            IF IS_RECORDING THEN
                RETURN 15
            ELSE
                RETURN 2
            END IF
        END METHOD
    END METHODS
END CLASS
