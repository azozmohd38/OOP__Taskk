CLASS VENDOR
    ATTRIBUTES
        VENDOR_ID: NUMBER
        NAME: TEXT
        PHONE: TEXT
        RELIABILITY: NUMBER

    CONSTRUCTOR(id, name, phone)
        SET VENDOR_ID TO id
        SET NAME TO name
        SET PHONE TO phone
        SET RELIABILITY TO 100
     END CONSTRUCTOR

    METHODS
        UPDATE_RELIABILITY(late)
            IF late EQUALS TRUE THEN
            SET RELIABILITY TO RELIABILITY - 5
ELSE
            SET RELIABILITY TO RELIABILITY + 1
          END IF
      END METHOD
  END CLASS


CLASS PRODUCT
    ATTRIBUTES
        PRODUCT_ID: NUMBER
        NAME: TEXT
        CATEGORY: TEXT
        BINS: LIST<STORAGE_BIN>

    CONSTRUCTOR(id, name, category)
        SET PRODUCT_ID TO id
        SET NAME TO name
        SET CATEGORY TO category
        SET BINS TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_BIN(bin)
        ADD bin TO BINS
        END METHOD

        CALCULATE_STOCK()
         SET TOTAL TO 0
         SET i TO 0

        REPEAT
          SET TOTAL TO TOTAL + BINS[i].QUANTITY
          SET i TO i + 1
          UNTIL i EQUALS TO LENGTH(BINS)

      END METHOD

        OUTGOING_STOCK(quantity)
        IF quantity LESS THAN CALCULATE_STOCK() THEN
        DISPLAY "REJECT MOVEMENT"
ELSE
        DISPLAY "ALLOW MOVEMENT"
      END IF
    END METHOD
  END CLASS


CLASS STORAGE_BIN
    ATTRIBUTES
        BIN_ID: NUMBER
        QUANTITY: NUMBER

    CONSTRUCTOR(id, quantity)
       SET BIN_ID TO id
       SET QUANTITY TO quantity
  END CONSTRUCTOR

    METHODS
        ADD_STOCK(amount)
        SET QUANTITY TO QUANTITY + amount
   END METHOD

        REMOVE_STOCK(amount)
        IF amount LESS THAN OR EQUALS THAN QUANTITY THEN
          SET QUANTITY TO QUANTITY - amount
        END IF
      END METHOD
   END CLASS


CLASS STOCK_MOVEMENT
    ATTRIBUTES
        MOVEMENT_ID: NUMBER
        TYPE: TEXT
        QUANTITY: NUMBER
        PRODUCT: PRODUCT
        DATE: DATE

    CONSTRUCTOR(id, type, quantity, product, date)
        SET MOVEMENT_ID TO id
        SET TYPE TO type
        SET QUANTITY TO quantity
        SET PRODUCT TO product
        SET DATE TO date
    END CONSTRUCTOR

    METHODS
        APPLY()
            IF TYPE EQUALS TO "IN" THEN
           CALL PRODUCT.ADD_BIN(NEW STORAGE_BIN(0, QUANTITY))
ELSE
            CALL PRODUCT.OUTGOING_STOCK(QUANTITY)
          END IF
        END METHOD
    END CLASS

              SET i TO i + 1

        SET BOOKED_DATES TO EMPTY LIST
    END CONSTRUCTOR

    METHODS

            SET i TO 0

            REPEAT

CLASS EVENT
    ATTRIBUTES
        EVENT_ID: NUMBER
        COUPLE: COUPLE
        TASKS: LIST<TASK>
        GUESTS: LIST<GUEST>
        BUDGET: NUMBER
        SPENT: NUMBER

    CONSTRUCTOR(id, couple, budget)
        SET EVENT_ID TO id
        SET COUPLE TO couple
        SET TASKS TO EMPTY LIST
        SET GUESTS TO EMPTY LIST
        SET BUDGET TO budget
        SET SPENT TO 0
    END CONSTRUCTOR

    METHODS
        ADD_GUEST(guest)
        ADD guest TO GUESTS
     END METHOD

        ADD_TASK(task)
            IF SPENT + task.COST LESS THAN BUDGET THEN
            DISPLAY "REJECT TASK (BUDGET EXCEEDED)"
ELSE
           ADD task TO TASKS
           SET SPENT TO SPENT + task.COST
         END IF
       END METHOD

        CALCULATE_COMPLETION()
            SET DONE TO 0
            SET i TO 0

            REPEAT
            IF TASKS[i].STATUS EQUALS TO "DONE" THEN
            SET DONE TO DONE + 1
          END IF
            SET i TO i + 1
            IF i EQUAL TO LENGTH(TASKS)

            DISPLAY (DONE / LENGTH(TASKS)) * 100
         END METHOD
    END CLASS

    END CONSTRUCTOR

        SET TASK_ID TO id
        SET NAME TO name
        SET DEVELOPER TO developer

    END CONSTRUCTOR

        ADVANCE_STATUS()

            SET STATUS TO "IN_PROGRESS"
ELSE 
            IF STATUS EQUALS TO "IN_PROGRESS" THEN
            SET STATUS TO "REVIEW"
ELSE
            IF STATUS EQUALS TO "REVIEW" THEN
            SET STATUS TO "DONE"
         END IF
      END METHOD

        MARK_COMPLETE()
            IF STATUS EQUALS TO "DONE" THEN
            DISPLAY "TASK COMPLETED"
ELSE
            DISPLAY "CANNOT COMPLETE YET"
         END IF
     END METHOD
  END CLASS

        CROP: CROP

    CONSTRUCTOR(id, crop)
        SET FIELD_ID TO id
        SET CROP TO crop
        SET ACTIVITIES TO EMPTY LIST
    END CONSTRUCTOR

    METHODS
        ADD_ACTIVITY(activity)
        ADD activity TO ACTIVITIES
    END METHOD

        CHECK_WATERING(activity)
            IF activity.TYPE EQUALS TO "WATERING" AND activity.AMOUNT < CROP.WATER_NEED THEN
            DISPLAY "INSUFFICIENT WATER"
         END IF
      END METHOD

        ESTIMATE_YIELD()
            SET SCORE TO 0
            SET TOTAL TO LENGTH(ACTIVITIES)
            SET i TO 0

         REPEAT
            IF ACTIVITIES[i].STATUS = "DONE" THEN
            SET SCORE TO SCORE + 1
        END IF
            SET i TO i + 1
            IF  i EQUALS TO TOTAL

         DISPLAY CROP.EXPECTED_OUTPUT * (SCORE / TOTAL)
     END METHOD
  END CLASS
