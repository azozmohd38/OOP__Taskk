                IF showtime.SCREEN.SCREEN_ID EQUALS SCREEN.SCREEN_ID THEN

                    IF START_TIME LESS THAN showtime.END_TIME AND
                       END_TIME GREATER THAN showtime.START_TIME THEN

            END FOR

            RETURN FALSE
        END METHOD

            ADD ticket TO TICKETS
        END METHOD

        END METHOD

CLASS STORAGE_BIN
    ATTRIBUTES
        BIN_ID: NUMBER

    CONSTRUCTOR(id: NUMBER, quantity: NUMBER)
        SET BIN_ID TO id
        SET QUANTITY TO quantity
    END CONSTRUCTOR


    METHODS

        ADD_STOCK(amount: NUMBER)
            SET QUANTITY TO QUANTITY + amount
        END METHOD


        REMOVE_STOCK(amount: NUMBER)

            IF amount LESS THAN OR EQUALS TO QUANTITY THEN
                SET QUANTITY TO QUANTITY - amount
            END IF

        END METHOD

    END METHODS
END CLASS

CLASS STOCK_MOVEMENT
    ATTRIBUTES
        MOVEMENT_ID: NUMBER
        TYPE: TEXT
        QUANTITY: NUMBER
        PRODUCT: PRODUCT
        DATE: DATE

    CONSTRUCTOR(id: NUMBER, type: TEXT, quantity: NUMBER, product: PRODUCT, date: DATE)
        SET MOVEMENT_ID TO id
        SET TYPE TO type
        SET QUANTITY TO quantity
        SET PRODUCT TO product
        SET DATE TO date
    END CONSTRUCTOR


    METHODS

        APPLY()

            IF TYPE EQUALS TO "IN" THEN

                IF LENGTH(PRODUCT.BINS) GREATER THAN 0 THEN
                    CALL PRODUCT.BINS[0].ADD_STOCK(QUANTITY)
                ELSE
                    CALL PRODUCT.ADD_BIN(NEW STORAGE_BIN(1, QUANTITY))
                END IF


            ELSE

                IF PRODUCT.OUTGOING_STOCK(QUANTITY) EQUALS TRUE THEN

                    SET REMAINING TO QUANTITY
                    SET i TO 0

                    REPEAT

                        IF PRODUCT.BINS[i].QUANTITY GREATER THAN OR EQUALS TO REMAINING THEN

                            CALL PRODUCT.BINS[i].REMOVE_STOCK(REMAINING)
                            SET REMAINING TO 0

                        ELSE

                            SET REMAINING TO REMAINING - PRODUCT.BINS[i].QUANTITY
                            CALL PRODUCT.BINS[i].REMOVE_STOCK(PRODUCT.BINS[i].QUANTITY)

                        END IF

                        SET i TO i + 1

                    UNTIL REMAINING EQUALS 0 OR i EQUALS LENGTH(PRODUCT.BINS)

                END IF

            END IF

        END METHOD

    END METHODS
END CLASS

CLASS STUDENT
    ATTRIBUTES
        STUDENT_ID: NUMBER
        NAME: TEXT
        CLASSES: LIST<CLASS_ROOM>
        GRADES: LIST<GRADE>


    CONSTRUCTOR(id: NUMBER, name: TEXT)
        SET STUDENT_ID TO id
        SET NAME TO name
        SET CLASSES TO EMPTY LIST
        SET GRADES TO EMPTY LIST
    END CONSTRUCTOR


    METHODS

        ENROLL(classRoom: CLASS_ROOM)

            SET CONFLICT TO FALSE

            FOR EACH enrolledClass IN CLASSES

                IF enrolledClass.SCHEDULE_TIME EQUALS classRoom.SCHEDULE_TIME THEN
                    SET CONFLICT TO TRUE
                END IF

            END FOR


            IF CONFLICT EQUALS FALSE THEN
                ADD classRoom TO CLASSES
                CALL classRoom.ENROLL_STUDENT(SELF)
            ELSE
                DISPLAY "SCHEDULE CONFLICT"
            END IF

        END METHOD


        ADD_GRADE(grade: GRADE)
            ADD grade TO GRADES
        END METHOD


        CALCULATE_AVERAGE()

            IF LENGTH(GRADES) EQUALS 0 THEN

                RETURN 0

            ELSE

                SET TOTAL TO 0
                SET WEIGHT_SUM TO 0
                SET i TO 0


                REPEAT

                    SET TOTAL TO TOTAL + (GRADES[i].MARK * GRADES[i].WEIGHT)
                    SET WEIGHT_SUM TO WEIGHT_SUM + GRADES[i].WEIGHT
                    SET i TO i + 1

                UNTIL i EQUALS LENGTH(GRADES)


                RETURN TOTAL / WEIGHT_SUM

            END IF

        END METHOD

    END METHODS
END CLASS

        SET FIELD_ID TO id
        SET CROP TO crop
        SET ACTIVITIES TO EMPTY LIST

    END CONSTRUCTOR



    METHODS


        ADD_ACTIVITY(activity: ACTIVITY)

            ADD activity TO ACTIVITIES

        END METHOD



        CHECK_WATERING(activity: ACTIVITY)

            IF activity.TYPE EQUALS "WATERING" AND activity.AMOUNT LESS THAN CROP.WATER_NEED THEN

                RETURN FALSE

            ELSE

                RETURN TRUE

            END IF

        END METHOD



        ESTIMATE_YIELD()

            IF LENGTH(ACTIVITIES) EQUALS 0 THEN

                RETURN 0

            ELSE

                SET SCORE TO 0
                SET TOTAL TO LENGTH(ACTIVITIES)
                SET i TO 0


                REPEAT

                    IF ACTIVITIES[i].STATUS EQUALS "DONE" THEN

                        SET SCORE TO SCORE + 1

                    END IF


                    SET i TO i + 1


                UNTIL i EQUALS TOTAL


                RETURN CROP.EXPECTED_OUTPUT * (SCORE / TOTAL)

            END IF

        END METHOD


    END METHODS
