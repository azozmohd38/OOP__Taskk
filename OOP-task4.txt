CLASS ATTEMPT
    ATTRIBUTES
        ATTEMPT_ID: NUMBER
        STUDENT: STUDENT
        EXAM: EXAM
        ANSWERS: LIST<ANSWER>
        START_TIME: TIME
        SCORE: NUMBER


    CONSTRUCTOR(id: NUMBER, student: STUDENT, exam: EXAM, startTime: TIME)

        SET ATTEMPT_ID TO id
        SET STUDENT TO student
        SET EXAM TO exam
        SET ANSWERS TO EMPTY LIST
        SET START_TIME TO startTime
        SET SCORE TO 0

        CALL student.ADD_ATTEMPT(SELF)

    END CONSTRUCTOR



    METHODS


        ADD_ANSWER(answer: ANSWER, currentTime: TIME)

            IF currentTime - START_TIME GREATER THAN EXAM.TIME_LIMIT THEN
            DISPLAY "TIME LIMIT EXCEEDED"
ELSE

           ADD answer TO ANSWERS
        END IF
     END METHOD



        CALCULATE_SCORE()
            SET TOTAL TO 0
            SET i TO 0
            REPEAT

            IF ANSWERS[i].SELECTED_ANSWER EQUALS EXAM.QUESTIONS[i].CORRECT_ANSWER THEN
            SET TOTAL TO TOTAL + EXAM.QUESTIONS[i].POINTS
        END IF
            SET i TO i + 1
            IF i EQUALS LENGTH(ANSWERS)
            RETURN TOTAL
        END METHOD
            GET_SCORE()
            SET SCORE TO CALCULATE_SCORE()
            RETURN SCORE
       END METHOD
   END METHODS
END CLASS



CLASS ANSWER
    ATTRIBUTES
        QUESTION: QUESTION
        SELECTED_ANSWER: TEXT


    CONSTRUCTOR(question: QUESTION, selected: TEXT)

        SET QUESTION TO question
        SET SELECTED_ANSWER TO selected

    END CONSTRUCTOR

END CLASS

CLASS MEMBER
    ATTRIBUTES
        MEMBER_ID: NUMBER
        NAME: TEXT
        PLAN: MEMBERSHIP_PLAN
        BOOKINGS: LIST<BOOKING>


    CONSTRUCTOR(id: NUMBER, name: TEXT, plan: MEMBERSHIP_PLAN)

        SET MEMBER_ID TO id
        SET NAME TO name
        SET PLAN TO plan
        SET BOOKINGS TO EMPTY LIST

    END CONSTRUCTOR



    METHODS


        ADD_BOOKING(booking: BOOKING)
        ADD booking TO BOOKINGS
     END METHOD
        CALCULATE_USED_HOURS()
        SET TOTAL TO 0
        SET i TO 0
        REPEAT
        SET TOTAL TO TOTAL + BOOKINGS[i].DURATION
        SET i TO i + 1
        IF i EQUALS LENGTH(BOOKINGS)
        RETURN TOTAL
     END METHOD
        REMAINING_HOURS()
        RETURN PLAN.MONTHLY_QUOTA - CALCULATE_USED_HOURS()
     END METHOD



        CAN_BOOK(hours: NUMBER)
            IF CALCULATE_USED_HOURS() + hours LESS THAN OR EQUAL TO PLAN.MONTHLY_QUOTA THEN
                RETURN TRUE
 ELSE
                RETURN FALSE
            END IF
        END METHOD
    END METHODS
END CLASS



CLASS ROOM
    ATTRIBUTES
        ROOM_ID: NUMBER
        NAME: TEXT
        BOOKINGS: LIST<BOOKING>


    CONSTRUCTOR(id: NUMBER, name: TEXT)

        SET ROOM_ID TO id
        SET NAME TO name
        SET BOOKINGS TO EMPTY LIST

    END CONSTRUCTOR



    METHODS


        IS_AVAILABLE(startTime: TIME, endTime: TIME)
        SET i TO 0


        REPEAT
        IF startTime LESS THAN BOOKINGS[i].END_TIME AND endTime GREATER THAN BOOKINGS[i].START_TIME THEN
        RETURN FALSE
      END IF
        SET i TO i + 1
        IF i EQUALS LENGTH(BOOKINGS)
        RETURN TRUE
     END METHOD
        ADD_BOOKING(booking: BOOKING)
        ADD booking TO BOOKINGS
    END METHOD
  END METHODS
END CLASS



CLASS BOOKING
    ATTRIBUTES
        BOOKING_ID: NUMBER
        MEMBER: MEMBER
        ROOM: ROOM
        START_TIME: TIME
        END_TIME: TIME
        DURATION: NUMBER


    CONSTRUCTOR(id: NUMBER, member: MEMBER, room: ROOM, startTime: TIME, endTime: TIME)

        SET BOOKING_ID TO id
        SET MEMBER TO member
        SET ROOM TO room
        SET START_TIME TO startTime
        SET END_TIME TO endTime
        SET DURATION TO endTime - startTime

    END CONSTRUCTOR



    METHODS


        CONFIRM_BOOKING()

            IF MEMBER.CAN_BOOK(DURATION) EQUALS TRUE AND ROOM.IS_AVAILABLE(START_TIME, END_TIME) EQUALS TRUE THEN
            CALL MEMBER.ADD_BOOKING(SELF)
            CALL ROOM.ADD_BOOKING(SELF)

                RETURN TRUE


ELSE

            RETURN FALSE
         END IF
      END METHOD
   END METHODS
END CLASS
