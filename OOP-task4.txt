CLASS ATTEMPT
    ATTRIBUTES
        ATTEMPT_ID: NUMBER
        STUDENT: STUDENT
        EXAM: EXAM
        ANSWERS: LIST<ANSWER>
        START_TIME: TIME
        SCORE: NUMBER

    CONSTRUCTOR(id: NUMBER, student: STUDENT, exam: EXAM, startTime: TIME)

        SET THIS.ATTEMPT_ID TO id
        SET THIS.STUDENT TO student
        SET THIS.EXAM TO exam
        SET THIS.ANSWERS TO EMPTY LIST
        SET THIS.START_TIME TO startTime
        SET THIS.SCORE TO 0

        CALL student.ADD_ATTEMPT(SELF)

    END CONSTRUCTOR

    METHODS

        ADD_ANSWER(answer: ANSWER, currentTime: TIME)

            IF currentTime - THIS.START_TIME GREATER THAN THIS.EXAM.TIME_LIMIT THEN
                DISPLAY "TIME LIMIT EXCEEDED"
            ELSE
                ADD answer TO THIS.ANSWERS
            END IF

        END METHOD

        CALCULATE_SCORE()

            SET TOTAL TO 0
            SET i TO 0

            REPEAT

                IF THIS.ANSWERS[i].SELECTED_ANSWER EQUALS THIS.EXAM.QUESTIONS[i].CORRECT_ANSWER THEN
                    SET TOTAL TO TOTAL + THIS.EXAM.QUESTIONS[i].POINTS
                END IF

                SET i TO i + 1

            UNTIL i = LENGTH(THIS.ANSWERS)

            RETURN TOTAL

        END METHOD

        GET_SCORE()

            SET THIS.SCORE TO CALCULATE_SCORE()
            RETURN THIS.SCORE

        END METHOD

    END METHODS
END CLASS



CLASS ANSWER
    ATTRIBUTES

        SELECTED_ANSWER: TEXT

    CONSTRUCTOR(question: QUESTION, selected: TEXT)

        SET THIS.QUESTION TO question
        SET THIS.SELECTED_ANSWER TO selected

    END CONSTRUCTOR

END CLASS

CLASS MEMBER
    ATTRIBUTES
        MEMBER_ID: NUMBER
        NAME: TEXT
        PLAN: MEMBERSHIP_PLAN
        BOOKINGS: LIST<BOOKING>


    CONSTRUCTOR(id: NUMBER, name: TEXT, plan: MEMBERSHIP_PLAN)

        SET THIS.MEMBER_ID TO id
        SET THIS.NAME TO name
        SET THIS.PLAN TO plan
        SET THIS.BOOKINGS TO EMPTY LIST

    END CONSTRUCTOR



    METHODS

        ADD_BOOKING(booking: BOOKING)

            ADD booking TO THIS.BOOKINGS

        END METHOD



        CALCULATE_USED_HOURS()

            SET TOTAL TO 0
            SET i TO 0

            REPEAT

                SET TOTAL TO TOTAL + THIS.BOOKINGS[i].DURATION
                SET i TO i + 1

            UNTIL i = LENGTH(THIS.BOOKINGS)

            RETURN TOTAL

        END METHOD



        REMAINING_HOURS()

            RETURN THIS.PLAN.MONTHLY_QUOTA - CALCULATE_USED_HOURS()

        END METHOD



        CAN_BOOK(hours: NUMBER)

            IF CALCULATE_USED_HOURS() + hours LESS THAN OR EQUAL TO THIS.PLAN.MONTHLY_QUOTA THEN

                RETURN TRUE

            ELSE

                RETURN FALSE

            END IF

        END METHOD


    END METHODS
END CLASS




CLASS ROOM
    ATTRIBUTES
        ROOM_ID: NUMBER
        NAME: TEXT
        BOOKINGS: LIST<BOOKING>


    CONSTRUCTOR(id: NUMBER, name: TEXT)

        SET THIS.ROOM_ID TO id
        SET THIS.NAME TO name
        SET THIS.BOOKINGS TO EMPTY LIST

    END CONSTRUCTOR



    METHODS


        IS_AVAILABLE(startTime: TIME, endTime: TIME)

            SET i TO 0

            REPEAT

                IF startTime LESS THAN THIS.BOOKINGS[i].END_TIME AND endTime GREATER THAN THIS.BOOKINGS[i].START_TIME THEN

                    RETURN FALSE

                END IF

                SET i TO i + 1

            UNTIL i = LENGTH(THIS.BOOKINGS)

            RETURN TRUE

        END METHOD



        ADD_BOOKING(booking: BOOKING)

            ADD booking TO THIS.BOOKINGS

        END METHOD


    END METHODS
END CLASS




CLASS BOOKING
    ATTRIBUTES
        BOOKING_ID: NUMBER
        MEMBER: MEMBER
        ROOM: ROOM
        START_TIME: TIME
        END_TIME: TIME
        DURATION: NUMBER


    CONSTRUCTOR(id: NUMBER, member: MEMBER, room: ROOM, startTime: TIME, endTime: TIME)

        SET THIS.BOOKING_ID TO id
        SET THIS.MEMBER TO member
        SET THIS.ROOM TO room
        SET THIS.START_TIME TO startTime
        SET THIS.END_TIME TO endTime
        SET THIS.DURATION TO endTime - startTime

    END CONSTRUCTOR



    METHODS


        CONFIRM_BOOKING()

            IF MEMBER.CAN_BOOK(DURATION) EQUALS TRUE AND ROOM.IS_AVAILABLE(START_TIME, END_TIME) EQUALS TRUE THEN

                CALL MEMBER.ADD_BOOKING(SELF)
                CALL ROOM.ADD_BOOKING(SELF)

                RETURN TRUE

            ELSE

                RETURN FALSE

            END IF

        END METHOD


    END METHODS
END CLASS
